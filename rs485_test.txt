void RS485Test_Thankyou_AI(void) {
    uint8_t test_msg[] = "RS485 TEST OK from STM32\r\n";
    uint8_t echo_msg[] = "ECHO: ";
    uint8_t rx_byte;
    uint8_t rx_data[64];
    uint8_t rx_index = 0;

    // Очищаем экран
    ssd1306_Fill(Black);

    // Шаг 1: Отправляем тестовое сообщение
    ssd1306_SetCursor(0, 0);
    ssd1306_WriteString("RS485 Test", Font_11x18, White);
    ssd1306_SetCursor(0, 20);
    ssd1306_WriteString("1. Sending test...", Font_6x8, White);
    ssd1306_UpdateScreen();

    HAL_GPIO_WritePin(GPIOA, GPIO_PIN_8, GPIO_PIN_SET); // TX mode
    HAL_Delay(2);
    HAL_UART_Transmit(&huart1, test_msg, sizeof(test_msg) - 1, 100);
    HAL_GPIO_WritePin(GPIOA, GPIO_PIN_8, GPIO_PIN_RESET); // RX mode
    HAL_Delay(2);

    // Шаг 2: Ждем ответ от пользователя
    ssd1306_SetCursor(0, 30);
    ssd1306_WriteString("2. Send me something!", Font_6x8, White);
    ssd1306_SetCursor(0, 40);
    ssd1306_WriteString("Type in terminal", Font_6x8, White);
    ssd1306_SetCursor(0, 50);
    ssd1306_WriteString("(10 sec timeout)", Font_6x8, White);
    ssd1306_UpdateScreen();

    // Ждем ответ 10 секунд
    uint32_t timeout = HAL_GetTick() + 10000; // 10 секунд
    rx_index = 0;

    while(HAL_GetTick() < timeout) {
        if(HAL_UART_Receive(&huart1, &rx_byte, 1, 50) == HAL_OK) {
            // Сохраняем принятый байт
            if(rx_index < sizeof(rx_data) - 1) {
                rx_data[rx_index++] = rx_byte;
            }

            // Показываем принятый символ на экране
            ssd1306_Fill(Black);
            ssd1306_SetCursor(0, 0);
            ssd1306_WriteString("RS485 Test", Font_11x18, White);
            ssd1306_SetCursor(0, 20);
            ssd1306_WriteString("RX received:", Font_6x8, White);
            ssd1306_SetCursor(0, 30);

            // Показываем последние 40 символов
            int start_idx = (rx_index > 40) ? rx_index - 40 : 0;
            for(int i = start_idx; i < rx_index; i++) {
                char display_char[2] = {rx_data[i], '\0'};
                ssd1306_WriteString(display_char, Font_6x8, White);
            }

            ssd1306_UpdateScreen();

            // Отправляем эхо-ответ
            HAL_GPIO_WritePin(GPIOA, GPIO_PIN_8, GPIO_PIN_SET);
            HAL_Delay(2);
            HAL_UART_Transmit(&huart1, echo_msg, 6, 100);
            HAL_UART_Transmit(&huart1, &rx_byte, 1, 100);
            uint8_t newline[] = "\r\n";
            HAL_UART_Transmit(&huart1, newline, 2, 100);
            HAL_GPIO_WritePin(GPIOA, GPIO_PIN_8, GPIO_PIN_RESET);
            HAL_Delay(2);
        }

        // Проверяем кнопку для досрочного выхода
        if(!HAL_GPIO_ReadPin(GPIOA, GPIO_PIN_3)) {
            break;
        }

        HAL_Delay(10);
    }

    // Шаг 3: Результаты
    ssd1306_Fill(Black);
    ssd1306_SetCursor(0, 0);
    ssd1306_WriteString("Test Results", Font_11x18, White);

    ssd1306_SetCursor(0, 20);
    ssd1306_WriteString("TX: OK (you saw text)", Font_6x8, White);

    ssd1306_SetCursor(0, 30);
    if(rx_index > 0) {
        char rx_msg[32];
        sprintf(rx_msg, "RX: OK (%d bytes)", rx_index);
        ssd1306_WriteString(rx_msg, Font_6x8, White);
    } else {
        ssd1306_WriteString("RX: No data received", Font_6x8, White);
    }

    ssd1306_SetCursor(0, 50);
    ssd1306_WriteString("Press any button...", Font_6x8, White);
    ssd1306_UpdateScreen();

    // Ждем нажатия любой кнопки
    while(1) {
        if(!HAL_GPIO_ReadPin(GPIOA, GPIO_PIN_0) ||
           !HAL_GPIO_ReadPin(GPIOA, GPIO_PIN_1) ||
           !HAL_GPIO_ReadPin(GPIOA, GPIO_PIN_2) ||
           !HAL_GPIO_ReadPin(GPIOA, GPIO_PIN_3) ||
           !HAL_GPIO_ReadPin(GPIOA, GPIO_PIN_4) ||
           !HAL_GPIO_ReadPin(GPIOA, GPIO_PIN_5)) {
            HAL_Delay(300);
            break;
        }
        HAL_Delay(50);
    }
}
